1  //=============================================================================
     2  // Testbench: alu_tb
     3  // Description: Coverage demonstration testbench for the 8-bit ALU
     4  //
     5  // Demonstrates:
     6  //   - Covergroups with sampling events
     7  //   - Coverpoints with explicit bins, auto-bins, and illegal_bins
     8  //   - Transition coverage (operation sequences)
     9  //   - Cross coverage (operation x operand ranges)
     10  //   - Constrained randomization
     11  //   - Coverage-driven stimulus (stop when coverage target met)
     12  //   - Self-checking with immediate assertions
     13  //=============================================================================
     14  
     15  module alu_tb;
     16  
     17      //=========================================================================
     18      // Signals
     19      //=========================================================================
     20      logic        clk;
     21      logic [7:0]  a, b;
     22      logic [2:0]  op;
     23      logic [7:0]  result;
     24      logic        zero, carry;
     25  
     26      //=========================================================================
     27      // DUT Instantiation
     28      //=========================================================================
     29      alu dut (
     30          .a      (a),
     31          .b      (b),
     32          .op     (op),
     33          .result (result),
     34          .zero   (zero),
     35          .carry  (carry)
     36      );
     37  
     38      //=========================================================================
     39      // Clock Generation: 10ns period
     40      //=========================================================================
     41      initial clk = 0;
     42      always #5 clk = ~clk;
     43  
     44      //=========================================================================
     45      // Coverage Definitions
     46      //=========================================================================
     47  
     48      // --- Covergroup: alu_cg ---
     49      // Sampled on the positive edge of clk
     50      covergroup alu_cg @(posedge clk);
     51  
     52          // ---- Coverpoint: ALU Operation ----
     53          // Named bins for every valid operation
     54          cp_op: coverpoint op {
     55              bins add_op = {3'b000};
     56              bins sub_op = {3'b001};
     57              bins and_op = {3'b010};
     58              bins or_op  = {3'b011};
     59              bins xor_op = {3'b100};
     60              bins not_op = {3'b101};
     61              bins sll_op = {3'b110};
     62              bins srl_op = {3'b111};
     63          }
     64  
     65          // ---- Coverpoint: Operand A (with interesting value bins) ----
     66          cp_a: coverpoint a {
     67              bins zero       = {8'h00};
     68              bins one        = {8'h01};
     69              bins mid_val    = {[8'h3F:8'h41]};  // around 0x40
     70              bins high_val   = {[8'hF0:8'hFE]};
     71              bins all_ones   = {8'hFF};
     72              bins low_range  = {[8'h02:8'h3E]};
     73              bins high_range = {[8'h42:8'hEF]};
     74          }
     75  
     76          // ---- Coverpoint: Operand B (with interesting value bins) ----
     77          cp_b: coverpoint b {
     78              bins zero       = {8'h00};
     79              bins one        = {8'h01};
     80              bins mid_val    = {[8'h3F:8'h41]};
     81              bins high_val   = {[8'hF0:8'hFE]};
     82              bins all_ones   = {8'hFF};
     83              bins low_range  = {[8'h02:8'h3E]};
     84              bins high_range = {[8'h42:8'hEF]};
     85          }
     86  
     87          // ---- Coverpoint: Result ranges ----
     88          cp_result: coverpoint result {
     89              bins res_zero    = {8'h00};
     90              bins res_low     = {[8'h01:8'h7F]};
     91              bins res_high    = {[8'h80:8'hFE]};
     92              bins res_max     = {8'hFF};
     93          }
     94  
     95          // ---- Coverpoint: Zero flag ----
     96          cp_zero_flag: coverpoint zero {
     97              bins flag_set   = {1'b1};
     98              bins flag_clear = {1'b0};
     99          }
     100  
     101          // ---- Coverpoint: Carry flag ----
     102          cp_carry_flag: coverpoint carry {
     103              bins carry_set   = {1'b1};
     104              bins carry_clear = {1'b0};
     105          }
     106  
     107          // ---- Transition Coverage on Operations ----
     108          // Track interesting operation sequences
     109          cp_op_trans: coverpoint op {
     110              bins add_to_sub    = (3'b000 => 3'b001);
     111              bins sub_to_add    = (3'b001 => 3'b000);
     112              bins arith_to_logic = (3'b000 => 3'b010),
     113                                    (3'b001 => 3'b010);
     114              bins logic_to_shift = (3'b010 => 3'b110),
     115                                    (3'b011 => 3'b110),
     116                                    (3'b100 => 3'b110);
     117              bins shift_to_arith = (3'b110 => 3'b000),
     118                                    (3'b111 => 3'b000);
     119              bins all_ops_seq    = (3'b000 => 3'b001 => 3'b010 => 3'b011 =>
     120                                     3'b100 => 3'b101 => 3'b110 => 3'b111);
     121          }
     122  
     123          // ---- Cross Coverage: Operation x Operand A ----
     124          // Ensures every operation is tested with every interesting 'a' range
     125          op_x_a: cross cp_op, cp_a;
     126  
     127          // ---- Cross Coverage: Operation x Operand B ----
     128          op_x_b: cross cp_op, cp_b;
     129  
     130          // ---- Cross Coverage: Operation x Zero Flag ----
     131          // Ensures each operation produces both zero and non-zero results
     132          op_x_zero: cross cp_op, cp_zero_flag;
     133  
     134      endgroup
     135  
     136      // Instantiate covergroup
     137      alu_cg cg = new();
     138  
     139      //=========================================================================
     140      // Stimulus Task: Apply inputs and wait one clock
     141      //=========================================================================
     142      task automatic apply_stimulus(input logic [2:0] t_op,
     143                                    input logic [7:0] t_a,
     144                                    input logic [7:0] t_b);
     145          @(negedge clk);  // Setup inputs on falling edge
     146          op = t_op;
     147          a  = t_a;
     148          b  = t_b;
     149          @(posedge clk);  // Covergroup samples here
     150          #1;              // Small delay for output to settle
     151      endtask
     152  
     153      //=========================================================================
     154      // Self-Checking: Verify ADD and SUB results
     155      //=========================================================================
     156      task automatic check_add(input logic [7:0] t_a, input logic [7:0] t_b);
     157          logic [8:0] expected;
     158          expected = {1'b0, t_a} + {1'b0, t_b};
     159          assert (result == expected[7:0])
     160              else $error("ADD FAIL: %0h + %0h = %0h, expected %0h",
     161                          t_a, t_b, result, expected[7:0]);
     162          assert (carry == expected[8])
     163              else $error("ADD CARRY FAIL: %0h + %0h carry=%0b, expected %0b",
     164                          t_a, t_b, carry, expected[8]);
     165      endtask
     166  
     167      task automatic check_sub(input logic [7:0] t_a, input logic [7:0] t_b);
     168          logic [8:0] expected;
     169          expected = {1'b0, t_a} - {1'b0, t_b};
     170          assert (result == expected[7:0])
     171              else $error("SUB FAIL: %0h - %0h = %0h, expected %0h",
     172                          t_a, t_b, result, expected[7:0]);
     173      endtask
     174  
     175      //=========================================================================
     176      // Main Test Stimulus
     177      //=========================================================================
     178      initial begin
     179          // Initialize signals
     180          op = 3'b000;
     181          a  = 8'h00;
     182          b  = 8'h00;
     183  
     184          $display("============================================");
     185          $display(" ALU Coverage Demonstration - Starting Test");
     186          $display("============================================");
     187  
     188          //---------------------------------------------------------------------
     189          // PHASE 1: Directed Tests (Corner Cases)
     190          //---------------------------------------------------------------------
     191          $display("\n--- Phase 1: Directed Corner-Case Tests ---");
     192  
     193          // ADD corner cases
     194          apply_stimulus(3'b000, 8'h00, 8'h00); check_add(8'h00, 8'h00);  // 0+0
     195          apply_stimulus(3'b000, 8'hFF, 8'h01); check_add(8'hFF, 8'h01);  // overflow
     196          apply_stimulus(3'b000, 8'hFF, 8'hFF); check_add(8'hFF, 8'hFF);  // max+max
     197          apply_stimulus(3'b000, 8'h80, 8'h80); check_add(8'h80, 8'h80);  // 128+128
     198          apply_stimulus(3'b000, 8'h01, 8'h01); check_add(8'h01, 8'h01);  // 1+1
     199  
     200          // SUB corner cases
     201          apply_stimulus(3'b001, 8'h00, 8'h00); check_sub(8'h00, 8'h00);  // 0-0
     202          apply_stimulus(3'b001, 8'hFF, 8'h01); check_sub(8'hFF, 8'h01);  // FF-1
     203          apply_stimulus(3'b001, 8'h00, 8'h01); check_sub(8'h00, 8'h01);  // underflow
     204          apply_stimulus(3'b001, 8'h40, 8'h40); check_sub(8'h40, 8'h40);  // equal
     205  
     206          // AND/OR/XOR with corner values
     207          apply_stimulus(3'b010, 8'hFF, 8'h00);  // AND: FF & 00 = 00
     208          apply_stimulus(3'b010, 8'hFF, 8'hFF);  // AND: FF & FF = FF
     209          apply_stimulus(3'b011, 8'h00, 8'h00);  // OR:  00 | 00 = 00
     210          apply_stimulus(3'b011, 8'hAA, 8'h55);  // OR:  AA | 55 = FF
     211          apply_stimulus(3'b100, 8'hFF, 8'hFF);  // XOR: FF ^ FF = 00
     212          apply_stimulus(3'b100, 8'hAA, 8'h55);  // XOR: AA ^ 55 = FF
     213  
     214          // NOT corner values
     215          apply_stimulus(3'b101, 8'h00, 8'h00);  // NOT 00 = FF
     216          apply_stimulus(3'b101, 8'hFF, 8'h00);  // NOT FF = 00
     217  
     218          // Shift corner values
     219          apply_stimulus(3'b110, 8'h80, 8'h00);  // SLL: 80 << 1 (carry out)
     220          apply_stimulus(3'b110, 8'h01, 8'h00);  // SLL: 01 << 1
     221          apply_stimulus(3'b111, 8'h01, 8'h00);  // SRL: 01 >> 1 (goes to 0)
     222          apply_stimulus(3'b111, 8'h80, 8'h00);  // SRL: 80 >> 1
     223  
     224          // Trigger the sequential transition: 000->001->010->011->100->101->110->111
     225          apply_stimulus(3'b000, 8'h12, 8'h34);
     226          apply_stimulus(3'b001, 8'h56, 8'h78);
     227          apply_stimulus(3'b010, 8'h9A, 8'hBC);
     228          apply_stimulus(3'b011, 8'hDE, 8'hF0);
     229          apply_stimulus(3'b100, 8'h11, 8'h22);
     230          apply_stimulus(3'b101, 8'h33, 8'h44);
     231          apply_stimulus(3'b110, 8'h55, 8'h66);
     232          apply_stimulus(3'b111, 8'h77, 8'h88);
     233  
     234          $display("Phase 1 Coverage: %.2f%%", cg.get_coverage());
     235  
     236          //---------------------------------------------------------------------
     237          // PHASE 2: Constrained-Random Stimulus
     238          //---------------------------------------------------------------------
     239          $display("\n--- Phase 2: Constrained-Random Stimulus (500 iterations) ---");
     240  
     241          repeat (500) begin
     242              logic [2:0] rand_op;
     243              logic [7:0] rand_a, rand_b;
     244  
     245              // Randomize
     246              assert(std::randomize(rand_op));
     247              assert(std::randomize(rand_a));
     248              assert(std::randomize(rand_b));
     249  
     250              apply_stimulus(rand_op, rand_a, rand_b);
     251  
     252              // Self-check for arithmetic operations
     253              if (rand_op == 3'b000) check_add(rand_a, rand_b);
     254              if (rand_op == 3'b001) check_sub(rand_a, rand_b);
     255          end
     256  
     257          $display("Phase 2 Coverage: %.2f%%", cg.get_coverage());
     258  
     259          //---------------------------------------------------------------------
     260          // PHASE 3: Coverage-Driven Closing Loop
     261          //---------------------------------------------------------------------
     262          $display("\n--- Phase 3: Coverage-Driven Closing Loop (target: 98%%) ---");
     263  
     264          begin
     265              automatic int iter_count = 0;
     266              while (cg.get_coverage() < 98.0 && iter_count < 2000) begin
     267                  logic [2:0] rand_op;
     268                  logic [7:0] rand_a, rand_b;
     269  
     270                  assert(std::randomize(rand_op));
     271                  assert(std::randomize(rand_a));
     272                  assert(std::randomize(rand_b));
     273  
     274                  apply_stimulus(rand_op, rand_a, rand_b);
     275                  iter_count++;
     276  
     277                  // Print progress every 200 iterations
     278                  if (iter_count % 200 == 0)
     279                      $display("  Iteration %0d, Coverage: %.2f%%",
     280                               iter_count, cg.get_coverage());
     281              end
     282              $display("  Closing loop finished after %0d iterations", iter_count);
     283          end
     284  
     285          //---------------------------------------------------------------------
     286          // Final Coverage Report
     287          //---------------------------------------------------------------------
     288          $display("\n============================================");
     289          $display(" FINAL COVERAGE RESULTS");
     290          $display("============================================");
     291          $display("  Overall Coverage:    %.2f%%", cg.get_coverage());
     292          $display("  cp_op Coverage:      %.2f%%", cg.cp_op.get_coverage());
     293          $display("  cp_a Coverage:       %.2f%%", cg.cp_a.get_coverage());
     294          $display("  cp_b Coverage:       %.2f%%", cg.cp_b.get_coverage());
     295          $display("  cp_result Coverage:  %.2f%%", cg.cp_result.get_coverage());
     296          $display("  cp_zero_flag:        %.2f%%", cg.cp_zero_flag.get_coverage());
     297          $display("  cp_carry_flag:       %.2f%%", cg.cp_carry_flag.get_coverage());
     298          $display("  cp_op_trans:         %.2f%%", cg.cp_op_trans.get_coverage());
     299          $display("  op_x_a Cross:        %.2f%%", cg.op_x_a.get_coverage());
     300          $display("  op_x_b Cross:        %.2f%%", cg.op_x_b.get_coverage());
     301          $display("  op_x_zero Cross:     %.2f%%", cg.op_x_zero.get_coverage());
     302          $display("============================================\n");
     303  
     304          #100;
     305          $finish;
     306      end
     307  
     308      //=========================================================================
     309      // Timeout Watchdog
     310      //=========================================================================
     311      initial begin
     312          #1_000_000;
     313          $display("ERROR: Simulation timeout!");
     314          $finish;
     315      end
     316  
     317  endmodule
