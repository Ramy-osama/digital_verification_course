1  //=============================================================================
     2  // Module: alu
     3  // Description: Simple 8-bit ALU for coverage demonstration
     4  //
     5  // Operations (op[2:0]):
     6  //   000 - ADD   : result = a + b
     7  //   001 - SUB   : result = a - b
     8  //   010 - AND   : result = a & b
     9  //   011 - OR    : result = a | b
     10  //   100 - XOR   : result = a ^ b
     11  //   101 - NOT   : result = ~a
     12  //   110 - SLL   : result = a << 1
     13  //   111 - SRL   : result = a >> 1
     14  //
     15  // Outputs:
     16  //   result[7:0] - ALU result
     17  //   zero        - Asserted when result == 0
     18  //   carry       - Carry/borrow output (valid for ADD/SUB)
     19  //=============================================================================
     20  
     21  module alu (
     22      input  logic [7:0] a,
     23      input  logic [7:0] b,
     24      input  logic [2:0] op,
     25      output logic [7:0] result,
     26      output logic       zero,
     27      output logic       carry
     28  );
     29  
     30      // Internal 9-bit result to capture carry
     31      logic [8:0] alu_wide;
     32  
     33      always_comb begin
     34          // Default values
     35          alu_wide = 9'b0;
     36  
     37          case (op)
     38              3'b000: begin // ADD
     39                  alu_wide = {1'b0, a} + {1'b0, b};
     40              end
     41  
     42              3'b001: begin // SUB
     43                  alu_wide = {1'b0, a} - {1'b0, b};
     44              end
     45  
     46              3'b010: begin // AND
     47                  alu_wide = {1'b0, a & b};
     48              end
     49  
     50              3'b011: begin // OR
     51                  alu_wide = {1'b0, a | b};
     52              end
     53  
     54              3'b100: begin // XOR
     55                  alu_wide = {1'b0, a ^ b};
     56              end
     57  
     58              3'b101: begin // NOT
     59                  alu_wide = {1'b0, ~a};
     60              end
     61  
     62              3'b110: begin // SLL (shift left logical by 1)
     63                  alu_wide = {a, 1'b0};  // MSB goes to carry
     64              end
     65  
     66              3'b111: begin // SRL (shift right logical by 1)
     67                  alu_wide = {1'b0, 1'b0, a[7:1]};  // LSB is lost
     68              end
     69  
     70              default: begin
     71                  alu_wide = 9'b0;
     72              end
     73          endcase
     74      end
     75  
     76      // Output assignments
     77      assign result = alu_wide[7:0];
     78      assign carry  = alu_wide[8];
     79      assign zero   = (alu_wide[7:0] == 8'b0);
     80  
     81  endmodule
